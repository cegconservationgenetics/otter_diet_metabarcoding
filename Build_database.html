#!/bin/bash

# Define gene regions search terms
declare -A gene_queries
gene_queries["12S"]='(12S[Title] OR 12s[Title] OR "s-rRNA"[Title] OR rrn12[Title] OR rrnS[Title] OR rrns[Title] OR srRNA[Title] OR "small subunit ribosomal RNA"[Title] OR "small ribosomal RNA subunit"[Title] OR mtDNA[Title] OR mitochondrial[Title] OR mitochondrion[Title] OR "mt-rRNA"[Title] OR "mt-rrnS"[Title])'
gene_queries["16S"]='(16S[Title] OR 16s[Title] OR rrn16[Title] OR rrnA[Title] OR rrnB[Title] OR lrRNA[Title] OR "l-rRNA"[Title] OR mtDNA[Title] OR mitochondrial[Title] OR mitochondrion[Title] OR "mt-rRNA"[Title] OR "mt-rrnL"[Title])'
gene_queries["COI"]='(COI[Title] OR CO1[Title] OR "cytochrome oxidase I"[Title] OR "cytochrome c oxidase subunit I"[Title] OR "cytochrome c oxidase I"[Title] OR cox1[Title] OR coxi[Title] OR "COX subunit 1"[Title] OR mtDNA[Title] OR mitochondrial[Title] OR mitochondrion[Title] OR "mt-co1"[Title] OR "mt-coI"[Title] OR "mt-rRNA"[Title])'

# Define taxonomic groups
groups=("Actinopterygii" "Amphibia" "Mammalia" "Aves" "Crocodylia" "Lepidosauria" "Testudines" "Insecta" "Bivalvia" "Gastropoda" "Clitellata" "Malacostraca" "Insecta")

# Process each gene region in sequence
for gene in "12S" "16S" "COI"; do
    gene_query="${gene_queries[$gene]}"

    # Process each taxonomic group
    for group in "${groups[@]}"; do
        echo "Processing $gene for $group..."

        qiime rescript get-ncbi-data \
            --p-query "$group[Organism] AND 80[SLEN]:30000[SLEN] AND ($gene_query) NOT \"environmental sample\"[Title] NOT \"environmental samples\"[Title] NOT environmental[Title] NOT uncultured[Title] NOT unclassified[Title] NOT unidentified[Title] NOT unverified[Title] NOT PREDICTED[Title]" \
            --p-ranks kingdom phylum class order family genus species \
            --p-rank-propagation \
            --p-n-jobs 5 \
            --o-sequences "${gene}_12smetazoa_sequence.qza" \
            --o-taxonomy "${gene}_12smetazoa_taxonomy.qza" \
            --verbose > "${gene}_12smetazoa_qiime_download.log" 2>&1

        # Add a small delay to avoid overwhelming the NCBI servers
        sleep 10
    done
done

#!/bin/bash

# Exit on first error
set -e

# Function to merge sequences
merge_sequences() {
    local input_dir=$1
    local output_dir=$2
    local prefix=$3
    local file_type=$4

    # Create output directory if it doesn't exist
    mkdir -p "$output_dir"

    # Check file type and use appropriate merge command
    if [ "$file_type" == "sequence" ]; then
        qiime feature-table merge-seqs \
            --i-data "$input_dir"/"$prefix"_*_"$file_type".qza \
            --o-merged-data "$output_dir"/"$prefix"_metazoa_"$file_type".qza
        echo "Merged $file_type for $prefix completed."
    elif [ "$file_type" == "taxonomy" ]; then
        qiime feature-table merge-taxa \
            --i-data "$input_dir"/"$prefix"_*_"$file_type".qza \
            --o-merged-data "$output_dir"/"$prefix"_metazoa_"$file_type".qza
        echo "Merged $file_type for $prefix completed."
    else
        echo "Invalid file type. Use 'sequence' or 'taxonomy'."
        exit 1
    fi
}

# Merge 16S rRNA sequences and taxonomy
merge_sequences "12s_db" "12smetazoa" "12S" "sequence"
merge_sequences "12s_db" "12smetazoa" "12S" "taxonomy"

# Merge 16S rRNA sequences and taxonomy
merge_sequences "16s_db" "16smetazoa" "16S" "sequence"
merge_sequences "16s_db" "16smetazoa" "16S" "taxonomy"

# Merge COI sequences and taxonomy
merge_sequences "coi_db" "coimetazoa" "COI" "sequence"
merge_sequences "coi_db" "coimetazoa" "COI" "taxonomy"

echo "All merging operations completed successfully."

#!/bin/bash
# Script to create a curated BLAST database

# Set variables
INPUT_SEQUENCES="/nas/collabs04/refseq/12smetazoa/12s_metazoa_sequence_derep1.qza"
INPUT_TAXONOMY="/nas/collabs04/refseq/12smetazoa/12s_metazoa_taxonomy_derep1.qza"
PRIMER_FWD="TTAGATACCCCACTATGC"
PRIMER_REV="TAGAACAGGCTCCTCTAG"
OUTPUT_FASTA="/nas/collabs04/refseq/12smetazoa/12s_12sv5_db.fasta"
BLAST_DB_DIR="/nas/collabs04/refseq/blastn/12s_12sv5_db"
BLAST_DB_NAME="12s_12sv5_db"
TAXID_MAP="12sv5_taxid_map.txt"

# Create the BLAST database directory if it doesn't exist
mkdir -p "$BLAST_DB_DIR"

# Load necessary tools and environments
eval "$(conda shell.bash hook)"
conda activate qiime2-amplicon-2023.9

# Check if activation was successful
if [ $? -ne 0 ]; then
    echo "Error: Failed to activate conda environment"
    exit 1
fi

echo "Step 1: Export sequences and taxonomy from QZA files"
# Create temporary working directory
WORK_DIR=$(mktemp -d)
cd "$WORK_DIR" || exit 1

# Check if input files exist
if [ ! -f "$INPUT_SEQUENCES" ] || [ ! -f "$INPUT_TAXONOMY" ]; then
    echo "Error: Input QZA files not found"
    exit 1
fi

mkdir -p exported_sequences exported_taxonomy

qiime tools export \
  --input-path "$INPUT_SEQUENCES" \
  --output-path "exported_sequences"

qiime tools export \
  --input-path "$INPUT_TAXONOMY" \
  --output-path "exported_taxonomy"

# Extract FASTA and taxonomy TSV files
FASTA_FILE="$WORK_DIR/exported_sequences/dna-sequences.fasta"
TAXONOMY_FILE="$WORK_DIR/exported_taxonomy/taxonomy.tsv"

# Check if files were exported successfully
if [ ! -f "$FASTA_FILE" ] || [ ! -f "$TAXONOMY_FILE" ]; then
    echo "Error: Failed to export QZA files"
    exit 1
fi

echo "Step 2: Bind sequences with primers using cutadapt"
BIND_FASTA="$WORK_DIR/bound_sequences.fasta"
cutadapt \
  -g "$PRIMER_FWD" \
  -a "$PRIMER_REV" \
  --discard-untrimmed \
  -o "$BIND_FASTA" \
  "$FASTA_FILE"

# Check if cutadapt was successful
if [ ! -f "$BIND_FASTA" ]; then
    echo "Error: Cutadapt failed to process sequences"
    exit 1
fi

echo "Step 3: Dereplicate sequences using vsearch"
DEREP_FASTA="$WORK_DIR/dereplicated_sequences.fasta"
vsearch \
  --derep_fulllength "$BIND_FASTA" \
  --output "$DEREP_FASTA" \
  --sizeout

# Check if vsearch was successful
if [ ! -f "$DEREP_FASTA" ]; then
    echo "Error: Vsearch failed to dereplicate sequences"
    exit 1
fi

echo "Step 4: Create taxid_map.txt"
TAXID_MAP_PATH="$BLAST_DB_DIR/$TAXID_MAP"

python3 <<EOF
import pandas as pd
import os
from tqdm import tqdm

try:
    print("Loading taxonomy data...")
    taxonomy = pd.read_csv("$TAXONOMY_FILE", sep='\t', header=0)
    print(f"Loaded {len(taxonomy)} taxonomy entries")

    print("Reading FASTA headers...")
    with open("$DEREP_FASTA", "r") as fasta_file:
        headers = [line.strip()[1:].split(';')[0] for line in fasta_file if line.startswith(">")]
    print(f"Found {len(headers)} sequence headers")

    print("Creating taxid map...")
    with open("$TAXID_MAP_PATH", "w") as map_file:
        for header in tqdm(headers, desc="Processing sequences"):
            tax_row = taxonomy[taxonomy['Feature ID'] == header]
            if not tax_row.empty:
                taxid = tax_row.index[0] + 1  # Use index+1 as taxid to ensure non-zero values
                map_file.write(f"{header}\t{taxid}\n")
            else:
                map_file.write(f"{header}\t1\n")  # Use 1 as default taxid

    print("Successfully created taxid map file")
    print(f"Output written to: {os.path.basename('$TAXID_MAP_PATH')}")
except Exception as e:
    print(f"Error creating taxid map: {str(e)}")
    exit(1)
EOF

echo "Step 5: Create the BLAST database"
# Copy dereplicated sequences to BLAST database directory
cp "$DEREP_FASTA" "$BLAST_DB_DIR/sequences.fasta"

# Check if copy was successful
if [ ! -f "$BLAST_DB_DIR/sequences.fasta" ]; then
    echo "Error: Failed to copy sequences file to BLAST database directory"
    exit 1
fi

# Change to the BLAST database directory
cd "$BLAST_DB_DIR" || exit 1

# Create BLAST database
makeblastdb \
  -in "sequences.fasta" \
  -dbtype nucl \
  -out "$BLAST_DB_NAME" \
  -parse_seqids \
  -taxid_map "$TAXID_MAP"

if [ $? -ne 0 ]; then
    echo "Error: makeblastdb failed"
    exit 1
fi

echo "Step 6: Clean up"
rm -rf "$WORK_DIR"

echo "BLAST database created successfully in: $BLAST_DB_DIR"
ls -l "$BLAST_DB_DIR"

#!/bin/bash

echo "Starting QIIME2 database building process at $(date)"

# Step 1: Convert sequences to DNA sequences artifact if needed (Derep 1)
echo "Using 12s_metazoa_sequence_derep1.qza as input for dereplication step 1..."
qiime tools export \
  --input-path "/nas/collabs04/refseq/12smetazoa/12s_metazoa_sequence_derep1.qza" \
  --output-path temp_seq

qiime tools import \
  --type "FeatureData[Sequence]" \
  --input-path temp_seq/dna-sequences.fasta \
  --output-path "/nas/collabs04/refseq/12smetazoa/12S_metazoa_sequence_derep1_DNA.qza"

# Step 2: Extract reads with primers (12SV5 primers)
echo "Extracting reads with primers (12sv5 F=TTAGATACCCCACTATGC, R=TAGAACAGGCTCCTCTAG)..."
qiime feature-classifier extract-reads \
    --i-sequences "/nas/collabs04/refseq/12smetazoa/12s_metazoa_sequence_derep1_DNA.qza" \
    --p-f-primer "TTAGATACCCCACTATGC" \
    --p-r-primer "TAGAACAGGCTCCTCTAG" \
    --p-identity 0.8 \
    --p-n-jobs 4 \
    --o-reads "/nas/collabs04/refseq/12smetazoa/12s_metazoa_extracted_reads.qza" \
    --verbose

# Step 3: Dereplicate sequences after primer extraction (Derep 2)
echo "Dereplicating sequences after primer extraction..."
qiime dereplicate \
  --i-sequences "/nas/collabs04/refseq/12smetazoa/12s_metazoa_extracted_reads.qza" \
  --o-dereplicated-sequences "/nas/collabs04/refseq/12smetazoa/12s_metazoa_sequence_derep2.qza" \
  --o-dereplicated-taxa "/nas/collabs04/refseq/12smetazoa/12s_metazoa_taxonomy_derep2.qza" \
  --verbose

# Step 4: Build the classifier
echo "Building the classifier..."
qiime feature-classifier fit-classifier-naive-bayes \
  --i-reference-reads "/nas/collabs04/refseq/12smetazoa/12s_metazoa_sequence_derep2.qza" \
  --i-reference-taxonomy "/nas/collabs04/refseq/12smetazoa/12s_metazoa_taxonomy_derep2.qza" \
  --o-classifier "/nas/collabs04/refseq/12smetazoa/12s_metazoa_12sv5_classifier.qza" \
  --verbose

echo "QIIME2 database building process completed at $(date)"
